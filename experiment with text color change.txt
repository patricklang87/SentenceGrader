<script>
let sentSamp = ["I", "went", "to", "the", "shop", ".", "and", "I", "bought", "things", "at", "the", "shop"];

let sentSampAr = [];
for (let i = 0; i < sentSamp.length; i++) {
	sentSampAr.push({word: sentSamp[i], status: 'normal'});
}

for (let j = 0; j < sentSampAr.length; j++) {
	if (sentSampAr[j].word == "shop") sentSampAr[j].status = "abnormal";
}



let expression = "<p>";
for (let k = 0; k < sentSampAr.length; k++) {
	if (sentSampAr[k].status == "abnormal") {
  	expression += "<span class='abnormal'>";
    expression += sentSampAr[k].word;
    expression = expression + " " + "</span>" ;
  }
  else expression += sentSampAr[k].word + " ";
}
expression.concat("</p>");
console.log(expression);


document.getElementById("exampleSentence").innerHTML = expression;
</script>
